    " expr = '#{expr.inspect}', is_eol = '#{is_eol.inspect}',"\
    " stack = #{stack.inspect}, stack_symbol = #{stack_symbol.inspect},"\

    " tokenized_line = #{@tokenized_line.inspect},"\
