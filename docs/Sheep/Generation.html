def add_child(obj)
  @__generation_children = [] if @__generation_children.nil?

  @__generation_children.push(obj)
  obj.parent = self
end

def top
  top_obj = self
  loop do
    break if top_obj.parent.nil?

    top_obj = top_obj.parent
  end

  return top_obj
end
