def check_exact_group_match(data)
  @exact_group_matches.each do |_, a_chain|
    test = MatchBase.check_exact_group_condition(a_chain, data)
    next if !test

    a_chain.init
    a_chain.matched(data)
    # a_chain.matched_end(data)
    return a_chain
  end
  return nil
end
