def matched(data)
  if @store_sym.nil?
    @store_sym = "_#{my_chain_num}".to_sym
  end

  if @matched_expr.instance_of?(Array)
    @matched_expr.push(data.expr)
  else
    @matched_expr = data.expr
  end
end

def reg_match(expr_, target_)
  # expr = expr_.gsub(/\||\?|\*|\(|\)|\{|\}|\[|\]|\+|\./) { |word| "\\#{word}" }
  # target = target_.gsub(/\||\?|\*|\(|\)|\{|\}|\[|\]|\+|\./) { |word| "\\#{word}" }
  rg = Regexp.new expr_
  @md = rg.match(target_)
  ldebug @md.inspect
  if !@md.nil?
    ldebug 'Found'
    return true
  else
    ldebug 'Not Found'
    return nil
  end
end
